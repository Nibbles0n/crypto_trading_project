Cond_EMA(x, cond, n) =>
    var float[] val = array.new_float()
    var float[] ema_val = array.new_float(1, na)
    if cond
        array.push(val, x)
        if array.size(val) > 1
            array.remove(val, 0)
        if na(array.get(ema_val, 0))
            array.fill(ema_val, array.get(val, math.max(0, array.size(val) - 1)))
        array.set(ema_val, 0, (array.get(val, 0) - array.get(ema_val, 0)) * (2 / (n + 1)) + array.get(ema_val, 0))
    array.get(ema_val, 0)

Cond_SMA(x, cond, n) =>
    var float[] vals = array.new_float()
    if cond
        array.push(vals, x)
        if array.size(vals) > n
            array.remove(vals, 0)
    array.avg(vals)

Stdev(x, n) =>
    math.sqrt(Cond_SMA(math.pow(x, 2), true, n) - math.pow(Cond_SMA(x, true, n), 2))

rng_size(x, scale, qty, n) =>
    ATR = Cond_EMA(ta.tr, true, n)  // ✅ v5: ta.tr()
    AC = Cond_EMA(math.abs(x - x[1]), true, n)
    SD = Stdev(x, n)
    switch scale
        "Pips" => qty * 0.0001
        "Points" => qty * syminfo.pointvalue
        "% of Price" => close * qty / 100
        "ATR" => qty * ATR
        "Average Change" => qty * AC
        "Standard Deviation" => qty * SD
        "Ticks" => qty * syminfo.mintick
        => qty  // "Absolute"

rng_filt(h, l, rng_, n, type, smooth, sn, av_rf, av_n) =>
    rng_smooth = Cond_EMA(rng_, true, sn)
    r = smooth ? rng_smooth : rng_
    var float[] rfilt = array.new_float(2, (h + l) / 2)
    array.set(rfilt, 1, array.get(rfilt, 0))
    if type == "Type 1"
        if h - r > array.get(rfilt, 1)
            array.set(rfilt, 0, h - r)
        if l + r < array.get(rfilt, 1)
            array.set(rfilt, 0, l + r)
    else if type == "Type 2"
        if h >= array.get(rfilt, 1) + r
            offset = math.floor(math.abs(h - array.get(rfilt, 1)) / r)
            array.set(rfilt, 0, array.get(rfilt, 1) + offset * r)
        if l <= array.get(rfilt, 1) - r
            offset = math.floor(math.abs(l - array.get(rfilt, 1)) / r)
            array.set(rfilt, 0, array.get(rfilt, 1) - offset * r)
    rng_filt1 = array.get(rfilt, 0)
    rng_filt2 = Cond_EMA(rng_filt1, rng_filt1 != rng_filt1[1], av_n)
    av_rf ? rng_filt2 : rng_filt1

// V5: ADX (cleaner)
calc_adx(len) =>
    up = ta.change(high)
    down = -ta.change(low)
    plusDM = na(up) ? na : up > down and up > 0 ? up : 0
    minusDM = na(down) ? na : down > up and down > 0 ? down : 0
    trur = ta.rma(ta.tr, len)
    plus = 100 * ta.rma(plusDM, len) / trur
    minus = 100 * ta.rma(minusDM, len) / trur
    sum = plus + minus
    100 * ta.rma(math.abs(plus - minus) / (sum == 0 ? 1 : sum), len)

            exit_price := close
            exit_reason := "Auto-Close (Same Dir)"
            same_direction_autocloses += 1

    // Explosive extension pause
    vol_spike_ex = volume > (vol_ma * explosive_vol_mult)
    price_accel = math.abs(close - close[3]) / close[3] * 100
    is_explosive_now = vol_spike_ex and price_accel >= explosive_accel
    filters_aligned = (filt1 > filt1[2]) == (filt2 > filt2[2])
    
    if enable_explosive_extension and is_explosive_now and filters_aligned and current_pnl_pct > 0
        if explosive_extension_pause == 0
            explosive_extensions += 1
        explosive_extension_pause := explosive_extension_bars
    if explosive_extension_pause > 0
        explosive_extension_pause -= 1

    // EXITS
    if not should_exit and use_profit_cap and current_pnl_pct >= regime_max_profit
        should_exit := true
        exit_price := close
        exit_reason := current_regime + " Max Profit"
        if current_regime == "RANGING"
            regime_ranging_exits += 1
        else if current_regime == "EXPLOSIVE" or current_regime == "PARABOLIC"
            regime_explosive_exits += 1

    if not should_exit and use_loss_cap and current_pnl_pct <= -max_loss_cap
        should_exit := true
        exit_price := close
        exit_reason := "Max Loss"

    if not should_exit and ((is_long and short_entry) or (not is_long and long_entry))
        should_exit := true
        exit_price := open
        exit_reason := "Opposite Signal"

    if exit_mode == "Signal + Peak Protection" and not should_exit and peak_protection_active and explosive_extension_pause == 0
        drawdown = ((profit_peak - current_pnl_pct) / profit_peak) * 100
        if drawdown >= regime_peak_dd and bars_since_peak >= peak_lookback_bars and current_pnl_pct > min_profit_threshold
            should_exit := true
            exit_price := close
            tier = current_pnl_pct < 8.0 ? "Small" : current_pnl_pct < 15.0 ? "Medium" : "Large"
            exit_reason := "Peak (" + tier + ")"

    // EXECUTE EXIT
    if should_exit
        exec_pnl = is_long ? (exit_price - active_entry) / active_entry * 100 : (active_entry - exit_price) / active_entry * 100
        net_pnl = exec_pnl * position_size_multiplier

        if parabolic_mode_active and exec_pnl > 0
            largest_parabolic_win := math.max(largest_parabolic_win, exec_pnl)

        total_trades += 1
        total_pnl_pct += net_pnl
        total_bars_in_trade += bars_in_trade

        if exec_pnl > 0
            winning_trades += 1
            largest_win := math.max(largest_win, exec_pnl)
        else
            losing_trades += 1
            largest_loss := math.min(largest_loss, exec_pnl)

        if show_exit_labels
            clr = net_pnl > 0 ? color.new(#00ff88, 0) : color.new(#ff3366, 0)
            pnl_str = str.tostring(math.round(net_pnl * 100) / 100, "#.##")
            size_str = position_size_multiplier != 1.0 ? "\n" + str.tostring(position_size_multiplier, "#.#") + "x" : ""
            badge = parabolic_mode_active ? " 🚀" : ""
            txt = exit_reason + badge + "\n" + pnl_str + "%" + size_str + "\n" + str.tostring(bars_in_trade) + " bars"
            label.new(bar_index, exit_price, txt, style=label.style_label_left, color=clr, textcolor=color.white, size=size.tiny)

        trade_active := false
        max_profit_pct := 0.0
        profit_peak := 0.0
        bars_since_peak := 0
        peak_protection_active := false
        explosive_extension_pause := 0
        position_size_multiplier := 1.0
        // No need to reset parabolic_mode_active — it’s var inside block

// ENTRY
if long_entry and not trade_active
    active_entry := open
    is_long := true
    trade_active := true
    entry_bar := bar_index
    entry_signal_rating_val := signal_rating
    position_size_multiplier := enable_signal_sizing ? 
      (signal_rating == 5 ? 1.5 : signal_rating == 4 ? 1.2 : signal_rating == 3 ? 0.7 : 0.5) : 1.0

if short_entry and not trade_active
    active_entry := open
    is_long := false
    trade_active := true
    entry_bar := bar_index
    entry_signal_rating_val := signal_rating
    position_size_multiplier := enable_signal_sizing ? 
      (signal_rating == 5 ? 1.5 : signal_rating == 4 ? 1.2 : signal_rating == 3 ? 0.7 : 0.5) : 1.0
